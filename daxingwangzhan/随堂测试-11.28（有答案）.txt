1.使用一条命令，创建/usr/local/tomcat/temp目录，并显示创建过程。

2.将/usr/local/java/jre/lib/management/jmxremote.password.template文件，重命名为jmxremote.password.
mv /usr/local/java/jre/lib/management/jmxremote.password.template /usr/local/java/jre/lib/management/jmxremote.password

3.查看/root下的所有文件的详细信息， 包括隐藏文件

4.为“fdisk -l  /dev/sdb”命令，自定义新的别名。

5.vim操作，复制从本行开始的三行，粘贴在光标之后，删除从本行开始的三行，从上而下查找str，从下而上查找str，查找下一个字符串，vi模式在第5行，读入/1.txt文件的内容。
	3yy	复制从本行开始的三行
	p	粘贴在光标之后
	3dd	删除从本行开始的三行
	/str	从上而下查找str
	?str	从下而上查找str
	n	查找下一个
	5（第五行）r /1.txt

18.使用fdisk分区格式化/dev/sdb，并挂载/sdb1分区至/mysql目录。
fdisk(gdisk) /dev/sdb	//分区
  mkfs.ext4 /dev/sdb1	//格式化
  mkfs.xfs  /dev/sdb1
  mkdir /mysql		//挂载
  mount /dev/sdb1 /mysql
  blkid 或者 df -hT	//查看挂载

19.将/dev/sdc磁盘配置为逻辑卷，格式化为xfs，并挂载至/mysql目录
pvcreate /dev/sdc
  vgcreate vg /dev/sdc
  lvcreate -L 200M -n lv vg
  mkfs.xfs /dev/vg/lv
  mount -t xfs /dev/vg/lv /mysql

20.逻辑卷/dev/vg1/lv2扩容 ,通过VG剩余空间20%比例来扩展LV2
 pvcreate /dev/sdd1
 vgextend vg1 /dev/sdd1
 lvextend -l 20%FREE /dev/VG/LV2

28.写出如下轮转规则的配置信息。日志文件：/var/log/yum.log，丢失不执行，空文件不论转，大小达到30k轮转，或者一年一轮转，延迟压缩，缩小周期到1天，轮转保留3次
 # vim /etc/logrotate.d/yum 
 /var/log/yum.log {
 missingok		//丢失不执行
 # notifempty		//空文件不论转
 # size 30k		//大小达到30k轮转,	 daily or  size两个维度
 # yearly		//或者一年一轮转
 # delaycompress //等会儿再压缩
 daily		//缩小周期到1天
 rotate 3		//轮转保留3次
 create 0777 root root
 }
